# Narrow grants for the leyline-server deploy flow. Reviewed 2026-05-13.
# Wildcards are sudoers-style globs (match any single path component).
#
# BEFORE INSTALLING: substitute `<operator-user>` with the SSH login user
# that runs the server deploy flow. Adjust the /tmp/leyline-server-* glob
# if your tempdir differs.
#
# Install with:
#   visudo -f /etc/sudoers.d/leyline-server

# Service control — used by deploy scripts and ad-hoc debugging.
<operator-user> ALL=(root) NOPASSWD: /sbin/rc-service leyline-server start, /sbin/rc-service leyline-server stop, /sbin/rc-service leyline-server restart, /sbin/rc-service leyline-server status

# Binary swap — used by deploy scripts. Source path is /tmp/leyline-server-<pid> only.
<operator-user> ALL=(root) NOPASSWD: /usr/bin/install -b -m 0755 -o leyline -g leyline /tmp/leyline-server-* /opt/leyline/bin/leyline-server

# Access-file write — used by install and bootstrap scripts. Source path is /tmp/leyline-access-<pid> only; destination is fixed.
<operator-user> ALL=(root) NOPASSWD: /usr/bin/install -m 0640 -o leyline -g leyline /tmp/leyline-access-* /opt/leyline/vaults/default/.leyline/vaultconfig/access
